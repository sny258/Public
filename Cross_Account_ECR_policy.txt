** To allow an ECR repo access to another role which is in another AWS account.
    After that that role can be attached to nodes, so that they can pull ECR repo images directly.

** Go to ECR repo and then open permissions. then edit the policy and add below json.


{  "Version": "2012-10-17",
  "Statement": [
    {
      "Sid": "AllowPushPull",
      "Effect": "Allow",
      "Principal": {
        "AWS": "arn:aws:iam::123456789:role/AmazonEKSFargatePodExecutionRole"
      },
      "Action": [
        "ecr:GetDownloadUrlForLayer",
        "ecr:BatchGetImage",
        "ecr:BatchCheckLayerAvailability",
        "ecr:PutImage",
        "ecr:InitiateLayerUpload",
        "ecr:UploadLayerPart",
        "ecr:CompleteLayerUpload"
      ]
    }
  ]
}

AND

{
